## Custom settings
# The icon path for the application logo
# Default: assets/resources/icons/minecraft.icns
icon:assets/resources/icons/minecraft.icns
# The title of the game
# Default: Minecraft++
title:Minecraft++
# Brand of the minecraft client (only used in F3 menu)
# Default: Minecraft++
brand:Minecraft++
# Wether or not the Minecraft++ brand should be send to the server
# If not, the client will tell a server, that it is a vanilla client
# It will disable client specific features, that are requested by a server like custom ui's
# Default: true
sendBrand:true
